/* Users can search champions by name dynamically without reloading webpage */
$('#text1').bind('input propertychange', function() {
	var name = $("#text1").val().toLowerCase(); 
	$('div.inner a.champion_icon').each(function() {
		$(this).show();
		if($(this).find("img").attr('id').toLowerCase().indexOf(name) != 0){
			$(this).hide();
		}
	});
});

/* Users can search items by name dynamically without reloading webpage */
$('#text2').bind('input propertychange', function() {
	var name = $("#text2").val().toLowerCase(); 
	var choice = $('input[name=radio]:checked').val();
	$('tbody tr').each(function() {
		$(this).show();
		if(choice == "name"){
			if($(this).find(":nth-child(2)").text().toLowerCase().indexOf(name) != 0){
				$(this).hide();
			}
		}
		else{
			if($(this).find(":nth-child(3)").text().toLowerCase().indexOf(name) < 0){
				$(this).hide();
			}
		}
	});
});				

<?php 

/* Parses and prints free champion rotation information from other fan website */

include_once('simple_html_dom.php');
require 'connect.php';

$html = new simple_html_dom();
$html->load_file('http://leagueoflegends.wikia.com/wiki/Free_champion_rotation');

$rotation = array();
foreach($html->find('ol[class=free_champion_rotation] li') as $element) {
    $spaces = array(" ", "&#32;");
    $champion = trim($element->plaintext);
    $champion = str_replace($spaces, "", $champion);
    array_push($rotation, $champion);
}

function print_free_rotation() {
    global $rotation;
    echo "<ul class='alt'>";
    foreach ($rotation as $i) {
        echo "<li>" . $i . "</li>";
    }
    echo "</ul>";
}

function print_this_week() {
    global $html;
    foreach ($html->find('div[id=rotationweek]') as $element) {
        echo $element->plaintext;
    }
}

function print_rotation_slide() {
    global $rotation;
    echo "<ul class='slideshow' width='600' height='350'>";
    foreach ($rotation as $i) {
        $src = "./img/skins/" . $i . ".jpg";
        echo "<li><img src='$src' width='600' height='350'/></li>";
    }
    
    echo "</ul>";
}

?>
